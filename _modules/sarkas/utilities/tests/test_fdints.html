[docs]def test_invfd1h():
    """Test the calculation of eta."""

    # Values taken from Tutorial Notebooks
    ne = 1.62e32  # [N/m^3]
    lambda_deB = 1.957093e-11  # [m]
    u = lambda_deB**3 * ne * sqrt(pi) / 4.0
    eta = invfd1h(u)

    assert isclose(eta, -0.28604631569655325)
